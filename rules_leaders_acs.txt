// src/rules/types.gleam
// This module defines the universal types for the Rules Engine.

// --- Core Types ---

pub type Source {
  FactionAbility(faction_id: String)
  Technology(tech_id: String)
  Anomaly(anomaly_type: Anomaly)
  AgendaLaw(agenda_id: String)
  Agent(agent_id: String)
  Commander(commander_id: String)
  Hero(hero_id: String)
  ActionCard(card_id: String)
}

pub type Rule {
  Rule(
    rule_id: String,
    source: Source,
    trigger: Trigger,
    effect: Effect,
  )
}

// --- Trigger Types ---
// Triggers define *when* a rule should be checked by the Rules Engine.

pub type Trigger {
  // General & Economic Triggers
  OnGameStart
  OnTransaction
  OnTechnologyResearch(by_opponent: Bool)
  OnGainControlOfPlanet(by_opponent: Bool)
  OnUnitDestroyed(is_opponent_unit: Bool, unit_type: UnitType)
  OnExploration(is_frontier: Bool)
  OnRelicGained
  PostProduction(in_home_system: Bool)
  OnSpendInfluence
  OnSpendResources

  // Action & Flow Triggers
  OnActionPhaseTurnStart
  OnActionCardPlayed(by_opponent: Bool)
  OnActionCardDrawn(by_opponent: Bool)
  OnStrategyCardSelection(by_opponent: Bool)
  OnStatusPhaseStart
  OnStatusPhaseScoreObjective(by_opponent: Bool)
  OnSystemActivation(by_opponent: Bool)
  OnTurnEnd

  // Movement & Board Triggers
  ValidateMovement
  OnRetreat
  PostMovement(into_system_with_own_units: Bool)

  // Combat Triggers
  PreCombatRound(is_attacker: Bool, is_defender: Bool)
  PreCombatRoll(is_attacker: Bool, is_defender: Bool)
  OnCombatRoll(is_opponent_roll: Bool)
  OnBombardmentRoll
  OnGroundCombatRoll(is_attacker: Bool, is_defender: Bool)
  OnInvasionStart(planet_id: String)
  OnSustainDamage(by_opponent: Bool)
  PostCombat(did_win: Bool, opponent_has_commander: Bool)

  // Agenda Phase Triggers
  OnAgendaRevealed
  OnAgendaPhaseStart
  OnAgendaVotingStart
  OnAgendaResolved
}

// --- Effect Types ---
// Effects define *what* a rule does when its trigger is met.

pub type Effect {
  // Player Stat Modifiers
  ModifyPlayerStat(stat: PlayerStat, modifier: Int)
  GainCommandToken

  // Combat Modifiers
  ModifyCombatRoll(modifier: Int)
  ModifyDiceResult(to_value: Int)
  GrantCombatRolls(unit_type: UnitType, count: Int)
  RerollDice(count: Int)
  DestroyUnit(unit_type: UnitType)
  DestroyOpponentUnit(unit_type: UnitType, count: Int)
  ModifySustainDamageRule(can_sustain_per_round: Bool)
  RemoveInfantry(count: Int)
  CancelHits(count: Int)
  ForceOpponentToAssignHits(count: Int)

  // Economic Modifiers
  StealTradeGoods(count: Int)
  GainTradeGoods(count: Int)
  GainCommodities(count: Int)
  ModifyPlanetSpend(discount: Int)
  ReadyPlanet(count: Int)
  ExhaustPlanet(is_opponent_planet: Bool)
  ForceOpponentToSpend(spend_type: SpendType, amount: Int)

  // Rule Overrides & Permissions
  IgnorePrerequisites(tech_colors: List(TechColor))
  IgnoreAdjacencyForTransactions
  IgnoreAnomaly(anomaly_type: Anomaly)
  IgnorePlacementRestriction(restriction: PlacementRestriction)
  IgnoreMecatolRexCost
  ModifyVotingRights(can_vote: Bool)
  GrantPdsSpaceCannonInSystem
  IgnorePlanetaryShield
  CancelActionCard
  CancelAgenda

  // Action Grants
  AddAction(action: Action)

  // Unit & Production Grants
  GrantAbilityToUnit(unit_type: UnitType, ability: UnitAbility)
  GrantProduction(unit_type: UnitType, count: Int)
  ModifyUnitStats(unit_type: UnitType, stats: List(UnitStatModifier))
  PlaceUnit(unit_type: UnitType)
  PlaceInfantry(count: Int)
  PlaceFighter(count: Int)
  CopyTechnology
  GainTechnology
  RepairUnit(count: Int)
  PreventOpponentProduction

  // Special Faction Effects
  ConvertOpponentInfantry
  SacrificeUnitForHits(unit_type: UnitType, count: Int)
  GainOpponentCommandToken
  CopyFactionAbility
  PlaceToken(token_type: TokenType)
  CaptureOpponentUnit(unit_type: UnitType)
  SwapStrategyCards
  DestroyPlanet
  GainRelic
  ScoreVictoryPoint
  LookAtTopCard(deck: DeckType)
  StealActionCard(count: Int)
  ForceOpponentToLoseVictoryPoint(count: Int)
}

// --- Supporting Enum Types ---

pub type PlayerStat {
  FleetPool
  CommandCounterCount
  Initiative
  ShipSpeed
}

pub type UnitType {
  Infantry
  Fighter
  Destroyer
  Cruiser
  Dreadnought
  WarSun
  Carrier
  PDS
  SpaceDock
  Structure // A generic type for PDS or Space Dock
  AnyShip
}

pub type UnitAbility {
  Bombardment
  SustainDamage
  Production
  SpaceCannon
  PlanetaryShield
}

pub type UnitStatModifier {
  ModifyCombat(modifier: Int)
  ModifyMove(modifier: Int)
  ModifyCost(modifier: Int)
  ModifyCapacity(modifier: Int)
}

pub type TechColor {
  Green
  Blue
  Yellow
  Red
}

pub type Anomaly {
  Nebula
  Supernova
  AsteroidField
}

pub type PlacementRestriction {
  SpaceDockMustBeOnPlanet
  PDSMustBeOnPlanet
}

pub type TokenType {
  Aetherstream
  Sleeper
  Wormhole
}

pub type DeckType {
  Action
  Agenda
  SecretObjective
}

pub type SpendType {
  Influence
  TradeGoods
  CommandToken
}

pub type Action {
  // Leader & Tech Actions
  AgentAction(agent_id: String)
  HeroAction(hero_id: String)
  // Action Card Actions
  ActionCardAction(card_id: String)
}
